00188 void DspComponent::StartAutoTick()
00189 {
00190   // Global scoped components (components not within a circuit) are added to the "global circuit" in
00191   // order to be auto-ticked. Technically it is only the global circuit that auto-ticks -This in turn
00192   // auto-ticks all components contained.
00193 
00194   // if this is the global circuit
00195   if( DSPatch::_IsThisGlobalCircuit( this ) )
00196   {
00197     if( _componentThread.IsStopped() )
00198     {
00199       _componentThread.Start();
00200 
00201       _isAutoTickRunning = true;
00202       _isAutoTickPaused = false;
00203     }
00204     else
00205     {
00206       ResumeAutoTick();
00207     }
00208   }
00209   // else if this component has no parent or it's parent is the global circuit
00210   else if( _parentCircuit == NULL || DSPatch::_IsThisGlobalCircuit( _parentCircuit ) )
00211   {
00212     DSPatch::_AddGlobalComponent( this );
00213     DSPatch::_StartGlobalAutoTick();
00214   }

00219 void DspComponent::StopAutoTick()
00220 {
00221   // If a component is part of the global circuit, a call to StopAutoTick() removes it from the
00222   // global circuit as to stop it from being auto-ticked. When all components are removed, the
00223   // global circuit auto-ticking is stopped.
00224 
00225   // if this is the global circuit
00226   if( DSPatch::_IsThisGlobalCircuit( this ) && !_componentThread.IsStopped() )
00227   {
00228     _componentThread.Stop();
00229 
00230     _isAutoTickRunning = false;
00231     _isAutoTickPaused = false;
00232   }
00233   // else if this component's parent is the global circuit
00234   else if( DSPatch::_IsThisGlobalCircuit( _parentCircuit ) )
00235   {
00236     DSPatch::_RemoveGlobalComponent( this );
00237 
00238     if( DSPatch::_GetGlobalComponentCount() == 0 )
00239     {
00240       DSPatch::_StopGlobalAutoTick();
00241     }
00242   }

00416 void DspComponent::_SetBufferCount( unsigned short bufferCount )
00417 {
00418   // _bufferCount is the current thread count / bufferCount is new thread count
00419 
00420   // delete excess _hasTickeds (if new buffer count is less than current)
00421   for( long i = _bufferCount - 1; i >= ( long ) bufferCount; i-- )
00422   {
00423     delete _hasTickeds[i];
00424   }
00425 
00426   // resize local buffer array
00427   _hasTickeds.resize( bufferCount );
00428 
00429   // create excess _hasTickeds (if new buffer count is more than current)
00430   for( unsigned short i = _bufferCount; i < bufferCount; i++ )
00431   {
00432     _hasTickeds[i] = new bool();
00433   }
00434 
00435   _inputBuses.resize( bufferCount );
00436   _outputBuses.resize( bufferCount );
00437 
00438   _gotReleases.resize( bufferCount );
00439   _releaseMutexes.resize( bufferCount );
00440   _releaseCondts.resize( bufferCount );
00441 
00442   for( unsigned short i = _bufferCount; i < bufferCount; i++ )
00443   {
00444     *_hasTickeds[i] = false;
00445     _gotReleases[i] = false;
00446 
00447     for( unsigned short j = 0; j < _inputBus.GetSignalCount(); j++ )
00448     {
00449       _inputBuses[i].AddSignal( _inputBus.GetSignal( j )->GetSignalName() );
00450     }
00451 
00452     for( unsigned short j = 0; j < _outputBus.GetSignalCount(); j++ )
00453     {
00454       _outputBuses[i].AddSignal( _outputBus.GetSignal( j )->GetSignalName() );

00456   }
00457 
00458   if( bufferCount > 0 )
00459   {

00475 void DspComponent::_ThreadTick( unsigned short threadNo )
00476 {
00477   // continue only if this component has not already been ticked
00478   if( *_hasTickeds[threadNo] == false )
00479   {
00480     // 1. set _hasTicked flag
00481     *_hasTickeds[threadNo] = true;
00482 
00483     // 2. get outputs required from input components
00484     for( unsigned short i = 0; i < _inputWires.GetWireCount(); i++ )
00485     {
00486       DspWire* wire = _inputWires.GetWire( i );
00487       wire->linkedComponent->_ThreadTick( threadNo );
00488 
00489       DspSignal* signal = wire->linkedComponent->_outputBuses[threadNo].GetSignal( wire->fromSignalIndex );
00490       _inputBuses[threadNo].SetSignal( wire->toSignalIndex, signal );
00491     }
00492 
00493     // 3. clear all outputs
00494     _outputBuses[threadNo].ClearAllValues();
00495 
00496     // 4. wait for your turn to process.
00497     _WaitForRelease( threadNo );
00498 
00499     // 5. call Process_() with newly aquired inputs
00500     Process_( _inputBuses[threadNo], _outputBuses[threadNo] );
00501 
00502     // 6. signal that you're done processing.
00503     _ReleaseThread( threadNo );
00504   }
00505 }
