00113 void DspCircuitThread::_Run()
00114 {
00115   if( _components != NULL )
00116   {
00117     while( !_stop )
00118     {
00119       _resumeMutex.Lock();
00120 
00121       _gotSync = true; // set the sync flag
00122 
00123       _syncCondt.WakeAll();
00124 
00125       if( !_gotResume ) // if haven't already got resume
00126       {
00127         _resumeCondt.Wait( _resumeMutex ); // wait for resume
00128       }
00129       _gotResume = false; // reset the resume flag
00130 
00131       _resumeMutex.Unlock();
00132 
00133       if( !_stop )
00134       {
00135         for( unsigned short i = 0; i < _components->size(); i++ )
00136         {
00137           ( *_components )[i]->_ThreadTick( _threadNo );
00138         }
00139         for( unsigned short i = 0; i < _components->size(); i++ )
00140         {
00141           ( *_components )[i]->_ThreadReset( _threadNo );
00142         }
00143       }
00144     }
00145   }
00146 
00147   _stopped = true;
00148 }
